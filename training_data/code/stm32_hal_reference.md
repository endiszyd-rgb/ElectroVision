# STM32 HAL (Hardware Abstraction Layer) Reference
# STM32F103 (Blue Pill), STM32F4, STM32L4 — CubeMX generated code patterns

## STM32F103C8 (Blue Pill) Key Specs
- ARM Cortex-M3 @ 72MHz
- 64-128KB Flash, 20KB SRAM
- 37 GPIO pins (PA0-PA15, PB0-PB15, PC13-PC15)
- 3× USART, 2× I2C, 2× SPI, 1× CAN, USB 2.0 FS
- 10× 12-bit ADC channels, 4× 16-bit timers
- Package: LQFP-48

## GPIO

```c
#include "stm32f1xx_hal.h"

// In gpio.c (generated by CubeMX):
void MX_GPIO_Init(void) {
    GPIO_InitTypeDef GPIO_InitStruct = {0};

    // Enable clocks FIRST!
    __HAL_RCC_GPIOC_CLK_ENABLE();
    __HAL_RCC_GPIOA_CLK_ENABLE();
    __HAL_RCC_GPIOB_CLK_ENABLE();

    // PC13 = Onboard LED (Blue Pill)
    HAL_GPIO_WritePin(GPIOC, GPIO_PIN_13, GPIO_PIN_SET);  // LED off (active low)
    GPIO_InitStruct.Pin = GPIO_PIN_13;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;           // Push-pull output
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
    HAL_GPIO_Init(GPIOC, &GPIO_InitStruct);

    // PA0 = Button input with pull-up
    GPIO_InitStruct.Pin = GPIO_PIN_0;
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);

    // PA8 = External interrupt (falling edge)
    GPIO_InitStruct.Pin = GPIO_PIN_8;
    GPIO_InitStruct.Mode = GPIO_MODE_IT_FALLING;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
    HAL_NVIC_SetPriority(EXTI9_5_IRQn, 0, 0);
    HAL_NVIC_EnableIRQ(EXTI9_5_IRQn);
}

// Usage in main.c:
HAL_GPIO_WritePin(GPIOC, GPIO_PIN_13, GPIO_PIN_RESET);   // LED on
HAL_GPIO_TogglePin(GPIOC, GPIO_PIN_13);                  // Toggle
uint8_t state = HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0);     // Read

// Interrupt handler in stm32f1xx_it.c:
void EXTI9_5_IRQHandler(void) {
    HAL_GPIO_EXTI_IRQHandler(GPIO_PIN_8);
}
// Callback called by HAL:
void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin) {
    if (GPIO_Pin == GPIO_PIN_8) {
        // Handle interrupt
    }
}
```

## USART (Serial Communication)

```c
// usart.c (generated):
void MX_USART1_UART_Init(void) {
    huart1.Instance = USART1;
    huart1.Init.BaudRate = 115200;
    huart1.Init.WordLength = UART_WORDLENGTH_8B;
    huart1.Init.StopBits = UART_STOPBITS_1;
    huart1.Init.Parity = UART_PARITY_NONE;
    huart1.Init.Mode = UART_MODE_TX_RX;
    huart1.Init.HwFlowCtl = UART_HWCONTROL_NONE;
    huart1.Init.OverSampling = UART_OVERSAMPLING_16;
    HAL_UART_Init(&huart1);
}

// Transmit (blocking)
uint8_t msg[] = "Hello STM32\r\n";
HAL_UART_Transmit(&huart1, msg, sizeof(msg)-1, HAL_MAX_DELAY);

// Receive (blocking)
uint8_t buf[32];
HAL_UART_Receive(&huart1, buf, 32, 1000);  // 1 second timeout

// Transmit via DMA (non-blocking)
HAL_UART_Transmit_DMA(&huart1, msg, sizeof(msg)-1);

// Printf via UART (retarget stdio)
int __io_putchar(int ch) {
    HAL_UART_Transmit(&huart1, (uint8_t *)&ch, 1, 0xFFFF);
    return ch;
}
```

```c
// i2c.c:
void MX_I2C1_Init(void) {
    hi2c1.Instance = I2C1;
    hi2c1.Init.ClockSpeed = 400000;             // 400kHz Fast mode
    hi2c1.Init.DutyCycle = I2C_DUTYCYCLE_2;
    hi2c1.Init.OwnAddress1 = 0;
    hi2c1.Init.AddressingMode = I2C_ADDRESSINGMODE_7BIT;
    hi2c1.Init.DualAddressMode = I2C_DUALADDRESS_DISABLE;
    hi2c1.Init.GeneralCallMode = I2C_GENERALCALL_DISABLE;
    hi2c1.Init.NoStretchMode = I2C_NOSTRETCH_DISABLE;
    HAL_I2C_Init(&hi2c1);
}

// Write to I2C device (register write)
uint8_t data[2] = {0xF4, 0x27};   // register, value
HAL_I2C_Master_Transmit(&hi2c1, 0x76 << 1, data, 2, HAL_MAX_DELAY);

// Read from I2C device (register read)
uint8_t reg = 0xF7;
HAL_I2C_Master_Transmit(&hi2c1, 0x76 << 1, &reg, 1, HAL_MAX_DELAY);
uint8_t buf[8];
HAL_I2C_Master_Receive(&hi2c1, 0x76 << 1, buf, 8, HAL_MAX_DELAY);

// Memory Read (combined write+read)
HAL_I2C_Mem_Read(&hi2c1, 0x76 << 1, 0xF7, I2C_MEMADD_SIZE_8BIT, buf, 8, HAL_MAX_DELAY);
```

```c
// Transmit + Receive
uint8_t tx[] = {0x80, 0x01};
uint8_t rx[2];
HAL_GPIO_WritePin(GPIOA, GPIO_PIN_4, GPIO_PIN_RESET);  // CS low
```

## Timers (TIM)

```c
// 1kHz timer interrupt (TIM2)
void MX_TIM2_Init(void) {
    htim2.Instance = TIM2;
    htim2.Init.Prescaler = 7200 - 1;       // 72MHz / 7200 = 10kHz
    htim2.Init.CounterMode = TIM_COUNTERMODE_UP;
    htim2.Init.Period = 10 - 1;             // 10kHz / 10 = 1kHz
    htim2.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
    HAL_TIM_Base_Init(&htim2);
    HAL_TIM_Base_Start_IT(&htim2);          // Start with interrupt
    HAL_NVIC_SetPriority(TIM2_IRQn, 0, 0);
    HAL_NVIC_EnableIRQ(TIM2_IRQn);
}

// PWM (TIM3_CH1 on PA6)
htim3.Init.Prescaler = 72 - 1;    // 1MHz timer clock
htim3.Init.Period = 1000 - 1;     // 1kHz PWM
HAL_TIM_PWM_Init(&htim3);
TIM_OC_InitTypeDef sConfig = {0};
sConfig.OCMode = TIM_OCMODE_PWM1;
sConfig.Pulse = 500;               // 50% duty cycle
HAL_TIM_PWM_ConfigChannel(&htim3, &sConfig, TIM_CHANNEL_1);
HAL_TIM_PWM_Start(&htim3, TIM_CHANNEL_1);

// Change duty cycle:
__HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, 750);  // 75%
```

## IWDG (Independent Watchdog)

```c
void MX_IWDG_Init(void) {
    hiwdg.Instance = IWDG;
    hiwdg.Init.Prescaler = IWDG_PRESCALER_64;  // 40kHz/64 = 625Hz
    hiwdg.Init.Reload = 625;                    // 625/625Hz = 1 second
    HAL_IWDG_Init(&hiwdg);
}

// In main loop — MUST call every <1 second:
HAL_IWDG_Refresh(&hiwdg);
```
